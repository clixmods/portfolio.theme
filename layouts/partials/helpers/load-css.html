{{/* Helper to load CSS file with automatic SCSS compilation, PostCSS, minify and fingerprint handling Usage: {{ partial "helpers/load-css.html" "css/main.css" }} Params: - . (string): Path to CSS file in assets/ Note: CSS files are compiled through SCSS pipeline (CSS is valid SCSS) Production mode: - Compiles through SCSS (for consistency) - Runs PostCSS (including PurgeCSS) to remove unused CSS - Minifies and fingerprints for cache busting - Adds integrity attribute for security */}} {{- $resource := resources.Get . }} {{- if $resource }} {{- with $resource | toCSS }} {{- if hugo.IsDevelopment }} {{- else }} {{- with . | css.PostCSS | minify | fingerprint | resources.PostProcess }} {{- end }} {{- end }} {{- end }} {{- else }} {{- errorf "CSS file not found: %s (check assets/ directory)" . }} {{- end }}
